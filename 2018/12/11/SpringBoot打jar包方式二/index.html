Spring boot配置pom打jar包--方式二

发表于 2018-12-11 更新于 2020-03-05 分类于 Spring Boot

方式二的打包方式和方式一的打包方式基本无恙，主要差别是指定了maven-assembly-plugin插件定制化打包

<build>
<finalName>cyam-lzh</finalName>
<plugins>
        <plugin>
            <groupId>org.springframework.boot</groupId>
            <artifactId>spring-boot-maven-plugin</artifactId>
        </plugin>    
        <!-- 使用该插件仅仅指定编译java时的版本-->
        <plugin>
            <groupId>org.apache.maven.plugins</groupId>  
            <artifactId>maven-compiler-plugin</artifactId>  
            <configuration> 
                <!-- 指定编译时使用的jdk版本--> 
                <source>1.8</source>  
                <target>1.8</target>  
            </configuration>  
        </plugin>
        <!-- 使用maven-assembly-plugin插件定制化打包-->
        <plugin>
            <artifactId>maven-assembly-plugin</artifactId>
            <configuration>
                <archive>
                    <manifest>
                        <mainClass>com.boot.Application</mainClass>
                    </manifest>
                </archive>
                <descriptorRefs>
                    <descriptorRef>
                        jar-with-dependencies
                    </descriptorRef>
                </descriptorRefs>
            </configuration>
        </plugin>    
    </plugins>
    <resources>
        <!--指定将src/main/resources/资源目录下的所有文件都打入 -->
        <resource>
            <directory>src/main/resources/</directory>
            <includes>
                <include>**/*.*</include>
            </includes>
        </resource>
    </resources>
  </build>

当然，你也可以指定使用另外一种打包插件来替换掉maven-assembly-plugin

<plugin>
    <groupId>org.apache.maven.plugins</groupId>
    <artifactId>maven-jar-plugin</artifactId>
    <configuration>
        <archive>
            <manifest>
                <addClasspath>true</addClasspath>
                <mainClass>com.boot.Application</mainClass>
                <classpathPrefix>
                    dependency/
                </classpathPrefix>
            </manifest>
        </archive>
    </configuration>
</plugin>

以上两种插件，根据个人喜好选择相应插件，但是无论选择哪种插件，
如果在项目中你引入了本地jar包，并且使用了《Maven 引入本地jar包的几种方式》
该教程中的方式二引入。你需要在resources标签内加入以下代码，来保证引入的本地jar包被成功的打入
最终的jar包中，如果是使用了方式一，则不需要加入以下代码。 

<resource>
    <!-- 指定打入jar包后的位置 -->
    <targetPath>lib/</targetPath>
    <!-- 本地jar包存放的资源目录 -->
    <directory>libs/</directory>
    <!-- 包含所有后缀为.jar的文件 -->
    <includes>
        <include>**/*.jar</include>
    </includes>
</resource>
